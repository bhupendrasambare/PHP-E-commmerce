CREATE TABLE `ecommerce`.`user` ( `id` INT(225) NOT NULL AUTO_INCREMENT ,  `uniqueid` INT(225) NOT NULL ,  `name` VARCHAR(225) NOT NULL ,  `email` VARCHAR(225) NOT NULL ,  `password` VARCHAR(225) NOT NULL ,  `address` VARCHAR(10000) NOT NULL ,    PRIMARY KEY  (`id`)) ENGINE = InnoDB;

CREATE TABLE `ecommerce`.`product` ( `id` INT NOT NULL AUTO_INCREMENT ,  `productid` INT(225) NOT NULL ,  `name` VARCHAR(225) NOT NULL ,  `subtitle` VARCHAR(225) NOT NULL ,  `company` VARCHAR(225) NOT NULL ,  `manager` VARCHAR(225) NOT NULL ,  `discription` VARCHAR(10000) NOT NULL ,  `orignal` INT(225) NOT NULL ,  `sale` INT(225) NOT NULL ,  `quantity` INT(225) NOT NULL ,    PRIMARY KEY  (`id`)) ENGINE = InnoDB;

CREATE TABLE `ecommerce`.`cart` ( `id` INT NOT NULL AUTO_INCREMENT ,  `userid` INT(225) NOT NULL ,  `productid` INT(225) NOT NULL ,  `quantity` INT(225) NOT NULL ,    PRIMARY KEY  (`id`)) ENGINE = InnoDB;
